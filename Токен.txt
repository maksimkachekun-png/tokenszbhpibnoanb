def gen_random_password():
    return ''.join(random.choices(string.ascii_letters + string.digits + "_@#%", k=12))

def mask_username(username: str) -> str:
    if not username:
        return "неизвестный"
    if len(username) <= 2:
        return username[0] + "*" * (len(username) - 1)
    return username[:2] + "*" * (len(username) - 4) + username[-2:]

def get_device_id_from_session(work_dir):
    db_path = os.path.join(work_dir, "session.db")
    for _ in range(10):
        if os.path.exists(db_path):
            try:
                conn = sqlite3.connect(db_path)
                cur = conn.cursor()
                cur.execute("SELECT * FROM sessions LIMIT 1")
                row = cur.fetchone()
                conn.close()
                if row and len(row) > 1:
                    return row[1]
            except:
                pass
        time.sleep(0.5)
    return ""

async def crypto_bot_create_invoice(amount):
    url = "https://pay.crypt.bot/api/createInvoice"
    headers = {
        "Content-Type": "application/json",
        "Crypto-Pay-API-Token": CRYPTO_BOT_TOKEN
    }
    payload = {
        "asset": "USDT",
        "amount": str(amount),
        "description": "Пополнение кассы FastSMS",
        "paid_btn_name": "callback",
        "paid_btn_url": "https://t.me/your_bot"
    }
    async with aiohttp.ClientSession() as session:
        resp = await session.post(url, headers=headers, json=payload)
        data = await resp.json()
        return data

async def check_subscription(user_id: int) -> bool:
    try:
        member = await bot.get_chat_member(REQUIRED_CHANNEL_ID, user_id)
        if member.status in ["left", "kicked"]:
            return False
    except Exception as e:
        logging.warning(f"Ошибка проверки подписки: {e}")
        return False
    return True

def get_all_accounts_list():
    all_accs = []
    for uid, accs in accounts.items():
        for phone, data in accs.items():
            all_accs.append({
                "uid": uid,
                "phone": phone,
                "token": data.get("token", ""),
                "user_id": data.get("user_id", 0),

            })
    return all_accs

def remove_account(uid, phone):
    if uid in accounts and phone in accounts[uid]:
        del accounts[uid][phone]
        if not accounts[uid]:
            del accounts[uid]

@dp.message(AuthState.waiting_withdraw_amount)
async def withdraw_amount(message: types.Message, state: FSMContext):
    try:
        amount = float(message.text.strip())
    except:
        return await message.answer("❌ Введите число.")

    stats = get_user_stats(message.from_user.id)
    if amount <= 0 or amount > stats['balance']:
        return await message.answer(f"❌ Сумма от 0.01 до {stats['balance']}.")

    uid = str(message.from_user.id)
    result = await crypto_bot_create_check(amount)

    logging.info(f"Crypto Bot check response: {result}")

    if result and result.get("ok"):
        res = result.get("result", {})
        check_url = res.get("check_url") or res.get("url") or res.get("link") or ""
        
        if not check_url:
            check_id = res.get("check_id", "")
            if check_id:
                check_url = f"https://t.me/CryptoBot?start=check_{check_id}"
        
        users[uid]["balance"] -= amount
        save_data()
        await log_withdraw(message.from_user.id, amount)
        
        if check_url:
            await message.answer(
                f"✅ Чек на <b>${amount}</b> создан!\n\n"
                f"Для получения нажмите на ссылку:\n{check_url}",
                parse_mode="HTML",
                disable_web_page_preview=True
            )
        else:
            await message.answer(
                f"✅ Чек на <b>${amount}</b> создан!\n\n"
                f"Проверьте в @CryptoBot",
                parse_mode="HTML"
            )
    else:
        error = result.get("error", {}) if result else {}
        error_name = error.get("name", "") if isinstance(error, dict) else ""
        
        if error_name == "NOT_ENOUGH_COINS":
            await message.answer("🏦 Казна бота пока что пуста, попробуйте позже.")
        else:
            error_text = error.get("name", "Неизвестная ошибка") if isinstance(error, dict) else str(error)
            await message.answer(f"❌ Ошибка: {error_text}")

    await state.clear()

@dp.callback_query(F.data == "admin_check_tokens")
async def cb_admin_check_tokens(callback: types.CallbackQuery):
    if callback.from_user.id != ADMIN_ID:
        return await callback.answer("❌ Только для админа", show_alert=True)

    all_accs = get_all_accounts_list()
    total = len(all_accs)

    if total == 0:
        await callback.answer("❌ Нет аккаунтов", show_alert=True)
        return

    await callback.answer("🔍 Проверка запущена...")
    
    msg = await callback.message.edit_text(f"🔍 Проверяю <b>{total}</b> токенов...", parse_mode="HTML")

    alive = 0
    dead = 0
    dead_list = []

    for acc in all_accs:
        valid = await check_token_valid(acc["phone"], acc["work_dir"])
        if valid:
            alive += 1
        else:
            dead += 1
            dead_list.append(acc)

    text = (
        f"🔍 <b>Результаты проверки</b>\n\n"
        f"✅ Живых: <b>{alive}</b>\n"
        f"❌ Мёртвых: <b>{dead}</b>\n"
        f"📱 Всего: <b>{total}</b>"
    )

    if dead > 0:
        pending[ADMIN_ID] = {"dead_tokens": dead_list}
        text += "\n\nНажмите кнопку чтобы удалить мёртвые токены."

    await msg.edit_text(text, parse_mode="HTML", reply_markup=token_check_keyboard() if dead > 0 else None)

        return

        count = int(message.text.strip())
    except:
        return await message.answer("❌ Введите число.")

    total = data.get("zip_total", 0)

    if count <= 0 or count > total:
        return await message.answer(f"❌ Введите число от 1 до {total}.")

    with zipfile.ZipFile(zip_buffer, 'w', zipfile.ZIP_DEFLATED) as zf:

    zip_buffer.seek(0)

    for acc in selected:
        remove_account(acc["uid"], acc["phone"])
    save_data()

    await bot.send_document(
        ADMIN_ID,
        BufferedInputFile(zip_buffer.getvalue(), filename=f"tokens_{count}.zip"),
        caption=f"📦 Выгружено {count} аккаунтов"
    )
    await message.answer(f"✅ Выгружено <b>{count}</b> аккаунтов. Они удалены из базы.", parse_mode="HTML")
    await state.clear()

@dp.message(AuthState.waiting_for_code, F.text.regexp(r'^\+7\d{10}$'))
async def phone_handler(message: types.Message, state: FSMContext):
    if not await ensure_subscribed(message):
        await state.clear()
        return
    phone = message.text.strip()
    uid = message.from_user.id
    work_dir = f"cache/{uid}/{phone}"
    os.makedirs(work_dir, exist_ok=True)

    db_path = os.path.join(work_dir, "session.db")
    if os.path.exists(db_path):
        os.remove(db_path)

    client = SocketMaxClient(phone=phone, work_dir=work_dir)
    pending[uid] = {"client": client, "phone": phone, "work_dir": work_dir, "status": "waiting_sms", "code_attempts": 0}
    await state.update_data(phone=phone, work_dir=work_dir)

    msg = await message.answer("⏳ Отправляю SMS...")

    async def auth():
        try:
            await client.start()
            await post_login(client, uid, phone, work_dir)
        except Exception as e:
            err = str(e).lower()
            if "attempt.limit" in err or "устарел" in err:
                if uid in pending and pending[uid].get("code_attempts", 0) < 2:
                    pending[uid]["code_attempts"] += 1
                    pending[uid]["status"] = "retry"
                else:
                    pending[uid]["status"] = "flood_sms"
            elif "not registered" in err or "not found" in err or "unregistered" in err:
                try:
                    reg_client = SocketMaxClient(phone=phone, work_dir=work_dir, registration=True)
                    await reg_client.start()
                    pending[uid]["client"] = reg_client
                    await post_login(reg_client, uid, phone, work_dir)
                    return
                except Exception as reg_err:
                    err = str(reg_err).lower()
                    if uid in pending:
                        if "block" in err: pending[uid]["status"] = "blocked"
                        elif "password" in err or "2fa" in err: pending[uid]["status"] = "need_password"
                        elif "limit" in err or "flood" in err: pending[uid]["status"] = "flood_sms"
                        elif "invalid" in err or "wrong" in err: pending[uid]["status"] = "invalid_code"
                        else: pending[uid]["status"] = f"error:{e}"
            else:
                if uid in pending:
                    if "block" in err: pending[uid]["status"] = "blocked"
                    elif "password" in err or "2fa" in err: pending[uid]["status"] = "need_password"
                    elif "limit" in err or "flood" in err: pending[uid]["status"] = "flood_sms"
                    elif "invalid" in err or "wrong" in err: pending[uid]["status"] = "invalid_code"
                    else: pending[uid]["status"] = f"error:{e}"

    async def post_login(client, uid, phone, work_dir):
        try:
            sessions = await client.get_sessions()
            current_id = None
            for sess in sessions:
                sess_dict = sess if isinstance(sess, dict) else sess.dict() if hasattr(sess, 'dict') else {}
                if sess_dict.get('current'):
                    current_id = sess_dict.get('id')
                    break
            for sess in sessions:
                sess_dict = sess if isinstance(sess, dict) else sess.dict() if hasattr(sess, 'dict') else {}
                if sess_dict.get('id') != current_id and sess_dict.get('id') is not None:
                    try:
                        await client.close_session(sess_dict['id'])
                    except:
                        pass
        except:
            pass

        new_password = gen_random_password()
        try:
            await client.change_2fa(new_password)
        except:
            try:
                await client.set_2fa(new_password)
            except:
                new_password = ""

        if uid in pending:
            pending[uid]["token"] = client.token
            pending[uid]["user_id"] = client.my_id
            if not pending[uid]["user_id"] and client.me:
                try:
                    me_dict = client.me.model_dump() if hasattr(client.me, 'model_dump') else client.me.dict()
                    pending[uid]["user_id"] = me_dict.get('contact', {}).get('id', 0)
                except: pass
            pending[uid]["password_2fa"] = new_password
            pending[uid]["status"] = "done"

    asyncio.ensure_future(auth())
    await asyncio.sleep(3)

    if uid not in pending: 
        return
    
    status = pending[uid].get("status", "")
    
    if status == "waiting_sms":
        await msg.edit_text("✅ SMS отправлена. Введите код из 6 цифр:")
    elif status == "retry":
        await msg.edit_text("📤 Отправляю новый код. Введите код из 6 цифр:")
    elif status == "flood_sms":
        await msg.edit_text("⏳ Слишком много попыток. Подождите и попробуйте позже.")
        del pending[uid]; await state.clear()
    elif status == "blocked":
        await msg.edit_text("🚫 Заблокирован.")
        del pending[uid]; await state.clear()
    elif status and status.startswith("error:"):
        await msg.edit_text(f"❌ {status[6:]}")
        del pending[uid]; await state.clear()
    else:
        await msg.edit_text("✅ SMS отправлена. Введите код из 6 цифр:")


@dp.message(AuthState.waiting_for_code, F.text.regexp(r'^\d{6}$'))
async def code_handler(message: types.Message, state: FSMContext):
    uid = message.from_user.id
    if uid not in pending: 
        return await message.answer("⏳ Сессия устарела. /start")

    data = await state.get_data()
    phone = data.get("phone", "")
    work_dir = data.get("work_dir", "")

    import builtins
    builtins._code = message.text.strip()

    msg = await message.answer("⏳ Проверяю код...")

    for _ in range(120):
        await asyncio.sleep(0.5)
        
        if uid not in pending:
            await msg.edit_text("⏳ Сессия устарела.")
            await state.clear()
            return
        
        status = pending[uid].get("status", "")

        if status == "retry":
            await msg.edit_text("🔄 Повторная попытка... Введите код ещё раз:")
            return

        if status == "done":
            token = pending[uid]["token"]
            vid = pending[uid].get("user_id", 0)
            pwd_2fa = pending[uid].get("password_2fa", "")
            uid_str = str(uid)
            
            if uid_str not in accounts: 
                accounts[uid_str] = {}
            accounts[uid_str][phone] = {
                "token": token,
                "user_id": vid,
                "work_dir": work_dir,
                "password": pwd_2fa
            }

            if uid_str not in users:
                users[uid_str] = {"balance": 0, "total_phones": 0, "total_earned": 0}
            users[uid_str]["total_phones"] += 1
            users[uid_str]["total_earned"] += PRICE_PER_PHONE
            users[uid_str]["balance"] += PRICE_PER_PHONE

            save_data()
            await msg.edit_text(f"✅ Номер принят! Аккаунт успешно встал\n💰 Начислено: ${PRICE_PER_PHONE}")
            del pending[uid]
            await state.clear()
            return

        elif status == "need_password":
            await state.set_state(AuthState.waiting_for_password)
            await msg.edit_text("🔐 Введите пароль 2FA:")
            return
        
        elif status == "flood_sms":
            await msg.edit_text("⏳ Слишком много попыток.")
            del pending[uid]
            await state.clear()
            return
        
        elif status == "invalid_code":
            await msg.edit_text("❌ Неверный код. Введите ещё раз:")
            return
        
        elif status == "blocked":
            await msg.edit_text("🚫 Заблокирован.")
            del pending[uid]
            await state.clear()
            return
        
        elif status and status.startswith("error:"):
            await msg.edit_text(f"❌ {status[6:]}")
            del pending[uid]
            await state.clear()
            return

    await msg.edit_text("❌ Таймаут. Попробуйте снова.")
    if uid in pending:
        del pending[uid]
    await state.clear()

@dp.message(AuthState.waiting_for_password)
async def password_handler(message: types.Message, state: FSMContext):
    uid = message.from_user.id
    if uid not in pending: return await message.answer("⏳ Сессия устарела. /start")

    data = await state.get_data()
    phone = data.get("phone", "")
    work_dir = data.get("work_dir", "")
    pwd = message.text.strip()

    import builtins
    builtins._password = pwd

    msg = await message.answer("⏳ Проверяю пароль...")
    await asyncio.sleep(5)

    db_path = os.path.join(work_dir, "session.db")
    token = None
    if os.path.exists(db_path):
        try:
            conn = sqlite3.connect(db_path)
            cur = conn.cursor()
            cur.execute("SELECT * FROM sessions LIMIT 1")
            row = cur.fetchone()
            if row:
                token = row[0]
                pending[uid]["token"] = token
                pending[uid]["status"] = "done"
            conn.close()
        except: pass

    if pending[uid].get("status") == "done":
        token = pending[uid].get("token", "")
        vid = pending[uid].get("user_id", 0)
        if not vid:
            client = pending[uid].get("client")
            if client and client.me:
                try:
                    me_dict = client.me.model_dump() if hasattr(client.me, 'model_dump') else client.me.dict()
                    vid = me_dict.get('contact', {}).get('id', 0)
                except: pass

        pwd_2fa = pending[uid].get("password_2fa", pwd)
        uid_str = str(uid)
        if uid_str not in accounts: accounts[uid_str] = {}
        accounts[uid_str][phone] = {"token": token, "user_id": vid, "work_dir": work_dir, "password": pwd_2fa}

        if uid_str not in users:
            users[uid_str] = {"balance": 0, "total_phones": 0, "total_earned": 0}
        users[uid_str]["total_phones"] += 1
        users[uid_str]["total_earned"] += PRICE_PER_PHONE
        users[uid_str]["balance"] += PRICE_PER_PHONE

        save_data()
        text = f"✅ Номер принят! Аккаунт успешно встал\n💰 Начислено: ${PRICE_PER_PHONE}"
        await msg.edit_text(text)
        del pending[uid]; await state.clear(); return

    await msg.edit_text("❌ Неверный пароль. Попробуйте ещё раз.")

async def main():
    import builtins, getpass
    oi, og = builtins.input, getpass.getpass
    def pi(prompt=""):
        p = prompt.lower()
        if "code" in p or "код" in p:
            while not hasattr(builtins,'_code') or builtins._code is None: time.sleep(0.5)
            c = builtins._code; builtins._code = None; return c
        return oi(prompt)
    def pg(prompt=""):
        p = prompt.lower()
        if "password" in p or "пароль" in p or "2fa" in p:
            for u in list(pending.keys()):
                if pending[u].get("status","") not in ("done","blocked","flood_sms"):
                    pending[u]["status"] = "need_password"
            while not hasattr(builtins,'_password') or builtins._password is None: time.sleep(0.5)
            pwd = builtins._password; builtins._password = None; return pwd
        return og(prompt)
    builtins.input = pi; getpass.getpass = pg
    logging.info("Бот запущен!")
    await dp.start_polling(bot)

if __name__ == "__main__":
    asyncio.run(main())
